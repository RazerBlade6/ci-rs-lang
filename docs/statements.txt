program        → statement* EOF ;

declaration    → varDecl
               | statement ;

               

statement      → exprStmt
               | printStmt
               | block ;

primary        → "true" | "false" | "nil"
               | NUMBER | STRING
               | "(" expression ")"
               | IDENTIFIER ;

exprStmt       → expression ";" ;
printStmt      → "print" expression ";" ;
varDecl        → "var" IDENTIFIER ( "=" expression )? ";" ;
block          → "{" declaration* "}" ;
