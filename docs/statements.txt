program        → statement* EOF ;

declaration    → varDecl
               | statement ;

               
statement      → exprStmt
               | ifStmt
               | printStmt
               | block ;

primary        → "true" | "false" | "nil"
               | NUMBER | STRING
               | "(" expression ")"
               | IDENTIFIER ;

exprStmt       → expression ";" ;
ifStmt         → "if" "(" expression ")" statement
               ( "else" statement )? ;
printStmt      → "print" expression ";" ;
varDecl        → "var" IDENTIFIER ( "=" expression )? ";" ;
block          → "{" declaration* "}" ;
